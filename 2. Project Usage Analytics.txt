SQL Analysis
Daily Active Users
SELECT
DATE(event_time) AS day,
COUNT(DISTINCT user_id) AS dau
FROM events
GROUP BY day;
Feature Usage
SELECT
feature_name,
COUNT(*) AS usage_count
FROM events
GROUP BY feature_name
ORDER BY usage_count DESC;
Retention Analysis
SELECT
signup_date,
COUNT(DISTINCT user_id)
FROM users
JOIN events USING(user_id)
GROUP BY signup_date;
